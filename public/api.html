User

// get all users
/api/users (GET)


// get single user by id
/api/userById/:id (GET)


// save user
/api/user/ (POST)
-param example:
{
	"name": "Cecil Stradford2",
	"phone": "+9728410621",
	"sms_enabled": true
}


// update user (PUT)
/api/user/:id
-param example:
{
	"name": "Cecil Stradford2",
	"phone": "+9728410621",
	"sms_enabled": true
}


// delete user (DELETE)
/api/user/:id
